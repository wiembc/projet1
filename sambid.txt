package info;

public class PC extends Produit {
	private double CPU;
	private int RAM;
	public PC(int reference, String marque, double CPU, int RAM) {
		super(reference, marque);
		this.CPU = CPU;
		this.RAM = RAM;
	}

	public String toString() {
		return "PC [cpu=" + CPU + ", ram=" + RAM + ", toString()=" + super.toString() + "]";
	} 

public Vend(String nom , String prenom , int age , double CA  , int NbU  ) {
		super ( nom , prenom , age , CA , NbU);
	}
	public  double CalculeSurface() {
		return (0.2*CA)+300;
		
	}
	
	
	public String toString() {
		return "le vendeur " +" " + super.toString()+" " +CalculeSurface() +"Dinars";

	}
}
